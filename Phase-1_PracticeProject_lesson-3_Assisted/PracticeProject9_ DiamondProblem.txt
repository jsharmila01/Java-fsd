interface first{
	default void display() {
		System.out.println("First method was shown");
	}
}
interface second {
	default void display() {
		System.out.println("Second Method was shown");
	}
}
class third implements first,second{
	public void display(){
		first.super.display();
		second.super.display();
	}
}
public class DiamondProblem {
	public static void main(String[] args) {
		third o = new third();
		o.display();
		
	}

}
