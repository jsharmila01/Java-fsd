class AccessSp
{ 
  void display() 
     { 
         System.out.println("default"); 
     } 
} 

public class accessSp1 {

	public static void main(String[] args) {
		System.out.println("This is Default access specifier");
		AccessSp obj = new AccessSp(); 		  
        obj.show(); 

	}
}
