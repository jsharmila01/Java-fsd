import java.util.Stack;

public class Stackdemo {

	public static void main(String[] args) {
		Stack s = new Stack();
		s.push(8);
		s.push(3);
		s.push(43);
		s.push(88);
		System.out.println("Stack before deletion: " +s);
		s.pop();
		s.pop();
		System.out.println("Stack after deletion: " +s);
		

	}

}
